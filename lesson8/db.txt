CREATE DATABASE IF NOT EXISTS pv311 CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE pv311;

CREATE TABLE IF NOT EXISTS tasks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    due DATE NULL,
    priority TINYINT NOT NULL CHECK (prioryty BETWEEN 1 AND 5),
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
